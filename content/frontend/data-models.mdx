# Data Models

TypeScript types used across the frontend. Models live under their feature area — `app/dashboard/models/`, `app/auth/models/`, and `app/shared/models/`.

---

## Invoice Types

```typescript
// app/dashboard/models/invoiceRead.ts
type InvoiceRead = {
  id: string;
  amount: number;
  date: Date;
  status: 'pending' | 'paid';
  customer: CustomerRead;

```

```typescript
// app/dashboard/models/invoiceCreate.ts
type InvoiceCreate = {
  status: string;

```

```typescript
// app/dashboard/models/invoiceUpdate.ts
type InvoiceUpdate = {
```

## Auth Types

### UserInfo

Returned from the OAuth2 login flow and UserInfo endpoint.

```typescript
// app/auth/models/user/userInfo.ts
type UserInfo = {
  id: string;
  email: string;
  profileImageUrl?: string | null;
  roleReads: Role[];
}
```

### UserAdminRead

Full admin view of a user account.

```typescript
// app/auth/models/user/userAdminRead.ts
type UserAdminRead = {
  id?: string;
  email?: string;
  emailVerified?: boolean;
  emailVerifiedAt?: string | null;
  locked?: boolean;
  failedLoginAttempts?: number;
  profileImageUrl?: string | null;
  roles: Role[];
  createdAt?: string | null;
  deletedAt?: string | null;
}
```

### UserSelfRead

A user's own account view (returned from `/api/v2/oauth2/userinfo` style self endpoints).

```typescript
// app/auth/models/user/userSelfRead.ts
type UserSelfRead = {
  id?: string;
  email?: string;
  emailVerified?: boolean;
  emailVerifiedAt?: string | null;
  profileImageUrl?: string | null;
  locked?: boolean;
}
```

## NextAuth Session

```typescript
// app/auth/models/next-auth.d.ts
declare module 'next-auth' {
  interface Session {
    accessToken: string;
    refreshToken: string;
    expiresAt: number;
    url: string;
    user: {
      id: string;
      role: Role[];
      grants: string[];   // flattened from roles
      image?: string | null;
    } & DefaultSession['user'];
  }

  interface User {
    id: string;
    email: string;
    role: Role[];
    grants: string[];
    accessToken: string;
    refreshToken: string;
    expiresIn: number;
    url: string;
    imageUrl: string | null;
  }
}

declare module 'next-auth/jwt' {
  interface JWT {
    id: string;
    role: Role[];
    grants: string[];
    accessToken: string;
    refreshToken: string;
    expiresAt: number;
    image: string | null;
  }
}
```
